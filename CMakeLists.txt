cmake_minimum_required(VERSION 3.14)
project(webview)

set(CMAKE_CXX_STANDARD 17)

if (${PORT} MATCHES "Cocoa")
    find_library(FOUNDATION Foundation)
    find_library(COCOA Cocoa)
    find_library(WEBKIT WebKit)

    add_executable(objc main.cpp webview.hpp webview.mm)
    add_executable(cpp main.cpp webview.hpp webview.cpp)

    target_link_libraries(objc ${FOUNDATION} ${COCOA} ${WEBKIT})
    target_link_libraries(cpp ${FOUNDATION} ${COCOA} ${WEBKIT})
    target_compile_options(objc PUBLIC -DPORT_COCOA)
    target_compile_options(cpp PUBLIC -DPORT_COCOA)
elseif (${PORT} MATCHES "GTK")
    add_executable(gtk main.cpp webview.hpp webview_gtk.cpp)
    if (${APPLE}) # Homebrew stuff
        set(ENV{PKG_CONFIG_PATH} "/usr/local/opt/libffi/lib/pkgconfig")
        target_include_directories(gtk PUBLIC /usr/local/opt/gettext/include)
        target_link_directories(gtk PUBLIC /usr/local/opt/gettext/lib)
    endif ()
    find_package(PkgConfig REQUIRED)
    pkg_check_modules(GTK REQUIRED gtk+-3.0)
    target_link_libraries(gtk ${GTK_LIBRARIES})
    target_include_directories(gtk PUBLIC ${GTK_INCLUDE_DIRS})
    target_compile_options(gtk PUBLIC ${GTK_CFLAGS_OTHER} -DPORT_GTK)
else ()
    message(WARNING "PORT is not defined. It should be \"Cocoa\" or \"GTK\"")
endif ()
